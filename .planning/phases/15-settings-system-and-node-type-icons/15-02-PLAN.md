<script>
(function () {
  'use strict';

  // Category switching
  window.showSettingsCategory = function (catId) {
    document.querySelectorAll('.settings-category-panel').forEach(function (p) {
      p.style.display = p.dataset.category === catId ? '' : 'none';
    });
    document.querySelectorAll('.settings-category-btn').forEach(function (b) {
      b.classList.toggle('active', b.dataset.category === catId);
    });
  };

  // Search filter: hide non-matching rows; hide sidebar buttons for fully-filtered categories
  window.filterSettings = function (query) {
    var q = query.toLowerCase().trim();
    document.querySelectorAll('.settings-row').forEach(function (row) {
      var text = (row.dataset.search || '').toLowerCase();
      var match = !q || text.includes(q);
      row.style.display = match ? '' : 'none';
    });
    // Hide categories where all rows are hidden
    document.querySelectorAll('.settings-category-panel').forEach(function (panel) {
      var catId = panel.dataset.category;
      var hasVisible = Array.from(panel.querySelectorAll('.settings-row'))
        .some(function (r) { return r.style.display !== 'none'; });
      var btn = document.querySelector('.settings-category-btn[data-category="' + catId + '"]');
      if (btn) btn.style.display = hasVisible ? '' : 'none';
    });
    // If the active category became hidden, activate the first visible one
    var activeBtn = document.querySelector('.settings-category-btn.active');
    if (activeBtn && activeBtn.style.display === 'none') {
      var firstVisible = document.querySelector('.settings-category-btn:not([style*="display: none"])');
      if (firstVisible) window.showSettingsCategory(firstVisible.dataset.category);
    }
  };

  // Instant apply: called by input onchange/oninput handlers
  window.settingChanged = function (key, value) {
    if (window.SemPKMSettings) window.SemPKMSettings.set(key, value);
  };

  // Reset single setting
  window.resetSingleSetting = function (key, btn) {
    if (!window.SemPKMSettings) return;
    window.SemPKMSettings.reset(key).then(function (data) {
      // Update the input to show the default value
      var row = btn ? btn.closest('.settings-row') : document.querySelector('[data-key="' + key + '"]');
      if (!row) return;
      var input = row.querySelector('input, select');
      if (input) {
        if (input.type === 'checkbox') {
          input.checked = data.default_value === 'true';
        } else {
          input.value = data.default_value || '';
        }
      }
      // Remove Modified badge and Reset button
      var badge = row.querySelector('.settings-modified-badge');
      if (badge) badge.remove();
      var resetBtn = row.querySelector('.settings-reset-btn');
      if (resetBtn) resetBtn.remove();
    });
  };

  // Reset all settings in a category
  window.resetCategorySettings = function (catId) {
    var panel = document.getElementById('category-' + catId);
    if (!panel) return;
    var rows = panel.querySelectorAll('.settings-row[data-key]');
    var promises = [];
    rows.forEach(function (row) {
      var key = row.dataset.key;
      var badge = row.querySelector('.settings-modified-badge');
      if (badge) {
        var resetBtn = row.querySelector('.settings-reset-btn');
        promises.push(window.resetSingleSetting(key, resetBtn));
      }
    });
  };

  // Re-init Lucide icons scoped to settings page
  if (typeof lucide !== 'undefined') {
    lucide.createIcons({ nodes: [document.getElementById('settings-page')] });
  }
})();
</script>

**Read `frontend/static/js/theme.js`** first to understand the current theme switching mechanism. The file currently reads from `localStorage` and exposes `window.setTheme()` and dispatches `sempkm:theme-changed`.

Add a `sempkm:setting-changed` listener that updates the theme when `core.theme` changes. This listener must be added AFTER the existing theme initialization logic (anti-FOUC is already handled by the inline script in base.html via localStorage — do not touch that):

At the end of `theme.js` (inside the IIFE, before the closing), add:

```javascript
// Settings system integration: react to sempkm:setting-changed for core.theme
document.addEventListener('sempkm:setting-changed', function (e) {
  if (e.detail && e.detail.key === 'core.theme') {
    var newTheme = e.detail.value;
    if (typeof window.setTheme === 'function') {
      window.setTheme(newTheme);
    }
    // Write-through: keep localStorage in sync so anti-FOUC fast-path is accurate
    try { localStorage.setItem('theme', newTheme); } catch (_) {}
  }
});
```

Also: when the user changes theme via the Settings page, `settingChanged('core.theme', value)` calls `SemPKMSettings.set(key, value)` which dispatches `sempkm:setting-changed`. This closes the loop: Settings page → SemPKMSettings.set → CustomEvent → theme.js listener → setTheme().

Additionally, ensure the theme.js initial load syncs with the server: after `DOMContentLoaded`, read `SemPKMSettings.get('core.theme')` and if different from what's currently applied (stored in `localStorage`), call `setTheme()`. Add this AFTER the settings listener, so settings.js has had time to populate its cache via auto-fetch:

```javascript
// On DOMContentLoaded, sync theme from server settings if different from localStorage fast-path
document.addEventListener('DOMContentLoaded', function () {
  // Small delay to let settings.js auto-fetch complete
  setTimeout(function () {
    var serverTheme = window.SemPKMSettings ? window.SemPKMSettings.get('core.theme') : null;
    if (serverTheme) {
      var localTheme;
      try { localTheme = localStorage.getItem('theme'); } catch (_) { localTheme = null; }
      if (serverTheme !== localTheme && typeof window.setTheme === 'function') {
        window.setTheme(serverTheme);
        try { localStorage.setItem('theme', serverTheme); } catch (_) {}
      }
    }
  }, 300);
});
```

This 300ms delay is acceptable: the anti-FOUC script already applied the localStorage theme before first paint; the sync only corrects if server and client disagree (rare — only happens on first login from a new browser or when server was updated while browser had stale localStorage).
