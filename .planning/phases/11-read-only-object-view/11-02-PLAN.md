**2. Add `toggleEditorMaximize` function in the object_tab.html script:**

Inside the existing IIFE (or in the deferred `_initEditMode_` closure), add:

```javascript
var _isMaximized = false;
var _savedSplitSizes = null;

window['toggleEditorMaximize'] = function(safeId) {
  var splitContainerId = 'object-split-' + safeId;
  var splitInstance = window._sempkmSplits && window._sempkmSplits[splitContainerId];
  if (!splitInstance) return;

  var btn = document.getElementById('maximize-btn-' + safeId);

  if (_isMaximized) {
    // Restore
    splitInstance.setSizes(_savedSplitSizes || [40, 60]);
    _isMaximized = false;
    if (btn) { btn.innerHTML = '&#9633;'; btn.title = 'Maximize Editor'; }
  } else {
    // Maximize
    _savedSplitSizes = splitInstance.getSizes();
    splitInstance.setSizes([0, 100]);
    _isMaximized = true;
    if (btn) { btn.innerHTML = '&#9635;'; btn.title = 'Restore Editor'; }
  }
};
```

Note: The `_isMaximized` and `_savedSplitSizes` variables should be scoped to the object tab. Since each tab has its own IIFE, these are naturally scoped. But since the function is assigned to `window`, use a per-tab key approach:

```javascript
window._editorMaxState = window._editorMaxState || {};

window.toggleEditorMaximize = function(safeId) {
  var splitContainerId = 'object-split-' + safeId;
  var splitInstance = window._sempkmSplits && window._sempkmSplits[splitContainerId];
  if (!splitInstance) return;

  var state = window._editorMaxState[safeId] || { maximized: false, savedSizes: null };
  var btn = document.getElementById('maximize-btn-' + safeId);

  if (state.maximized) {
    splitInstance.setSizes(state.savedSizes || [40, 60]);
    state.maximized = false;
    if (btn) { btn.innerHTML = '&#9633;'; btn.title = 'Maximize Editor'; }
  } else {
    state.savedSizes = splitInstance.getSizes();
    splitInstance.setSizes([0, 100]);
    state.maximized = true;
    if (btn) { btn.innerHTML = '&#9635;'; btn.title = 'Restore Editor'; }
  }

  window._editorMaxState[safeId] = state;
};
```
