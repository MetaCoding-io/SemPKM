    Create backend/app/vfs/auth.py implementing `SemPKMWsgiAuthenticator`. This class validates Basic auth credentials in the WSGI thread pool — it MUST use synchronous SQLAlchemy (not async), because wsgidav is a WSGI app.

    ```python
    import hashlib
    from wsgidav.dc.base_dc import BaseDomainController

    class SemPKMWsgiAuthenticator(BaseDomainController):
        """wsgidav domain controller that validates API tokens via sync SQLAlchemy."""

        def __init__(self, wsgidav_app, config: dict) -> None:
            super().__init__(wsgidav_app, config)
            # Extract db_url from wsgidav config; strip async driver prefix
            db_url = config.get("sempkm_db_url", "")
            self._sync_db_url = db_url.replace("+aiosqlite", "")

        def get_domain_realm(self, path_info: str, environ: dict) -> str:
            return "SemPKM"

        def require_authentication(self, realm: str, environ: dict) -> bool:
            return True

        def is_share_anonymous(self, share_path: str) -> bool:
            return False

        def basic_auth_user(self, realm: str, user_name: str, password: str, environ: dict) -> bool:
            """Validate username (email) + password (API token). Returns True if valid."""
            from sqlalchemy import create_engine, select
            from sqlalchemy.orm import Session
            from app.auth.models import ApiToken, User

            token_hash = hashlib.sha256(password.encode()).hexdigest()
            engine = create_engine(self._sync_db_url)
            try:
                with Session(engine) as session:
                    result = session.execute(
                        select(ApiToken).join(User).where(
                            User.email == user_name,
                            ApiToken.token_hash == token_hash,
                            ApiToken.revoked_at.is_(None),
                        )
                    )
                    token_row = result.scalar_one_or_none()
                    if token_row:
                        # Store user info in environ for downstream use
                        environ["sempkm.user_email"] = user_name
                        return True
                    return False
            finally:
                engine.dispose()
    ```

    Create backend/app/vfs/provider.py implementing `SemPKMDAVProvider`:

    ```python
    from wsgidav.dav_provider import DAVProvider
    from app.triplestore.sync_client import SyncTriplestoreClient

    class SemPKMDAVProvider(DAVProvider):
        """Routes WebDAV path requests to collection/resource instances."""

        def __init__(self, sync_client: SyncTriplestoreClient) -> None:
            super().__init__()
            self._client = sync_client

        def get_resource_inst(self, path: str, environ: dict):
            """Dispatch path to appropriate DAV resource.

            Path structure:
              /                          -> RootCollection
              /{model-id}/               -> ModelCollection
              /{model-id}/{type-label}/  -> TypeCollection
              /{model-id}/{type-label}/{filename}.md -> ResourceFile
            """
            from app.vfs.collections import RootCollection, ModelCollection, TypeCollection
            from app.vfs.resources import ResourceFile

            parts = [p for p in path.strip("/").split("/") if p]

            if len(parts) == 0:
                return RootCollection("/", environ, self._client)
            elif len(parts) == 1:
                return ModelCollection(path, environ, self._client, model_id=parts[0])
            elif len(parts) == 2:
                return TypeCollection(path, environ, self._client, model_id=parts[0], type_label=parts[1])
            elif len(parts) == 3 and parts[2].endswith(".md"):
                return ResourceFile(path, environ, self._client, model_id=parts[0], type_label=parts[1], filename=parts[2])
            return None
    ```

    Create backend/app/vfs/collections.py with three collection classes. All inherit from `wsgidav.dav_provider.DAVCollection`.

    **Directory structure implemented:**
    - `/dav/` → RootCollection: lists model IDs (SPARQL: find distinct model IDs from installed models)
    - `/dav/{model-id}/` → ModelCollection: lists type labels in that model
    - `/dav/{model-id}/{type-label}/` → TypeCollection: lists .md files (one per object)

    **SPARQL queries to use (all scoped to `FROM <urn:sempkm:current>`):**

    RootCollection — find installed models:
    ```sparql
    PREFIX sempkm: <https://sempkm.org/ontology/>
    SELECT DISTINCT ?modelId FROM <urn:sempkm:current>
    WHERE { ?model a sempkm:MentalModel ; sempkm:modelId ?modelId . }
    ```

    ModelCollection — find type labels in a model:
    ```sparql
    PREFIX sempkm: <https://sempkm.org/ontology/>
    PREFIX sh: <http://www.w3.org/ns/shacl#>
    SELECT DISTINCT ?typeLabel FROM <urn:sempkm:current>
    WHERE {
      ?model a sempkm:MentalModel ; sempkm:modelId "{model_id}" .
      ?shape sh:targetClass ?class .
      BIND(REPLACE(STR(?class), ".*/", "") AS ?typeLabel)
    }
    ```

    TypeCollection — find objects of a type, return label and IRI:
    ```sparql
    PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
    PREFIX dcterms: <http://purl.org/dc/terms/>
    PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
    PREFIX foaf: <http://xmlns.com/foaf/0.1/>
    SELECT ?iri ?label FROM <urn:sempkm:current>
    WHERE {
      ?iri a <{type_iri}> .
      OPTIONAL { ?iri dcterms:title ?t }
      OPTIONAL { ?iri rdfs:label ?r }
      OPTIONAL { ?iri skos:prefLabel ?s }
      OPTIONAL { ?iri foaf:name ?f }
      BIND(COALESCE(?t, ?r, ?s, ?f, REPLACE(STR(?iri), ".*/", "")) AS ?label)
    }
    ORDER BY ?label
    ```

    **Filename generation:** `slugify(label)` — lowercase, replace non-alphanumeric with `-`, deduplicate hyphens. If two objects produce the same slug, append `--{short_iri_hash}` (first 6 chars of SHA-256 of IRI) to disambiguate.

    **Read-only enforcement:** Override `is_locked` and handle_delete/handle_move in DAVCollection to raise `DAVError(HTTP_FORBIDDEN)`. Do NOT implement any write methods.

    The TypeCollection must store a mapping of `filename → (iri, label, type_iri)` so ResourceFile can look up the object without re-querying SPARQL.

    For TypeCollection, store the type_iri by looking it up from SHACL shapes. For MVP, accept the type_label as the local name of the type IRI and reconstruct by querying shapes. Alternatively, store the full type IRI in the TypeCollection itself (pass it in from the parent query).

            """Return {"iri": ..., "label": ..., "type_iri": ...} or None."""
