# Plan 23-02: Custom YASR cell renderer for SemPKM IRIs, dark mode overrides

## Goal

Extend the Yasgui SPARQL console (from Plan 23-01) with two features:

1. **Custom YASR cell renderer** — SemPKM object IRIs in query results render as styled clickable `<a>` links that open the object in the active editor group (calls `window.openTab(iri, label)`).
2. **Dark mode CSS overrides** — Yasgui's UI matches the workspace theme in both light and dark modes using SemPKM CSS custom properties.

## Context

YASR (Yet Another SPARQL Resultset viewer) is Yasgui's result rendering component. It supports a plugin system where custom cell renderers can intercept specific values and return custom HTML.

**YASR plugin API pattern:**
```js
Yasgui.YASR.plugins.myPlugin = {
  hideFromSelection: true,
  priority: 10,
  canHandleResults: function (yasr) { return true; },
  draw: function (yasr, plugin) {
    // access results via yasr.results
  }
};
```

However, for cell-level customization, YASR uses a table plugin with value formatter support. The correct hook is to register a custom **value formatter** on the Table plugin that transforms URI values matching SemPKM IRIs into HTML anchor elements.

**YASR Table value hook (confirmed API):**
```js
Yasgui.YASR.plugins.table.defaults.persistentConfig = false;
// Register a formatter that wraps matching cells
var originalFormatter = Yasgui.YASR.plugins.table.defaults.uriFormatter;
Yasgui.YASR.plugins.table.defaults.uriFormatter = function (uri, compact) {
  if (isSemPKMIri(uri)) {
    return renderSemPKMLink(uri);
  }
  return originalFormatter ? originalFormatter(uri, compact) : null;
};
```

**IRI detection strategy:**
- SemPKM object IRIs are created with the configured base namespace (workspace objects have `https://` scheme and contain the deployment base URL)
- Safe heuristic: detect any IRI that is NOT a well-known vocabulary namespace (`rdf:`, `rdfs:`, `owl:`, `dcterms:`, `skos:`, `foaf:`, `schema:`, `xsd:`) and has an `https://` scheme
- Avoid false positives: do not link to external URLs (W3C, schema.org, etc.) — only link IRIs that would resolve to SemPKM objects
- Better approach: check if the IRI would match the `/browser/object/{iri}` route by testing against the server's base URL — but since we cannot know the base URL at template time without a template variable, use a data attribute on the container: `<div id="yasgui-container" data-base-url="{{ request.base_url }}" ...>`

**IRI link behavior (from CONTEXT.md):**
- Clicking a SemPKM IRI opens the object in the active editor group
- Same pattern as `openTab()` used by nav tree — `window.openTab(iri, label)`
- `label` for the link text: use the IRI's last path segment as a short form, or the full IRI if no path segment is available

**Dark mode strategy:**
- Yasgui uses its own CSS with class selectors (`.yasgui`, `.yasqe`, `.yasr`)
- SemPKM uses `[data-theme="dark"]` attribute on `<html>`
- Override Yasgui variables inside a `[data-theme="dark"] .yasgui { }` block in `theme.css`
- Map Yasgui's colors to SemPKM's existing CSS custom properties

### Implement custom YASR URI cell renderer

In the inline script block from Plan 23-01 (in `workspace.html`), after the `initYasgui` function definition and before `window.initYasguiConsole = initYasgui`, add the custom cell renderer registration:

```js
// -----------------------------------------------------------------------
// Custom YASR URI cell renderer for SemPKM object IRIs
// -----------------------------------------------------------------------

function getSemPKMBaseUrl() {
  // Read base URL from container data attribute (set in template)
  var container = document.getElementById('yasgui-container');
  if (container && container.dataset.baseUrl) {
    return container.dataset.baseUrl.replace(/\/$/, '');
  }
  // Fallback: use current origin
  return window.location.origin;
}

var KNOWN_VOCAB_PREFIXES = [
  'http://www.w3.org/1999/02/22-rdf-syntax-ns#',
  'http://www.w3.org/2000/01/rdf-schema#',
  'http://www.w3.org/2002/07/owl#',
  'http://purl.org/dc/terms/',
  'http://purl.org/dc/elements/1.1/',
  'http://www.w3.org/2004/02/skos/core#',
  'http://xmlns.com/foaf/0.1/',
  'https://schema.org/',
  'http://schema.org/',
  'http://www.w3.org/2001/XMLSchema#',
  'http://www.w3.org/ns/shacl#',
  'http://www.w3.org/ns/prov#',
  'http://www.w3.org/ns/dcat#',
  'urn:sempkm:',
  'tag:rdf4j.org,'
];

function isSemPKMObjectIri(uri) {
  // Must be an https:// IRI
  if (!uri || uri.indexOf('https://') !== 0) return false;

  // Reject well-known vocabulary namespaces
  for (var i = 0; i < KNOWN_VOCAB_PREFIXES.length; i++) {
    if (uri.indexOf(KNOWN_VOCAB_PREFIXES[i]) === 0) return false;
  }

  // Must contain a path segment (not just a domain)
  try {
    var parsed = new URL(uri);
    if (parsed.pathname === '/' || parsed.pathname === '') return false;
  } catch (e) {
    return false;
  }

  return true;
}

function shortenIri(uri) {
  try {
    var parsed = new URL(uri);
    var parts = parsed.pathname.split('/').filter(Boolean);
    if (parts.length > 0) {
      var last = decodeURIComponent(parts[parts.length - 1]);
      // Limit display length
      if (last.length > 40) last = last.substring(0, 37) + '...';
      return last;
    }
  } catch (e) {}
  // Fallback: truncate the full IRI
  return uri.length > 50 ? uri.substring(0, 47) + '...' : uri;
}

function registerYasrUriFormatter() {
  // Yasgui.YASR.plugins.table exists on the Yasgui global
  if (!window.Yasgui || !Yasgui.YASR || !Yasgui.YASR.plugins || !Yasgui.YASR.plugins.table) {
    return false; // Not loaded yet
  }

  var tablePlugin = Yasgui.YASR.plugins.table;
  if (!tablePlugin.defaults) tablePlugin.defaults = {};

  // Store reference to original formatter if any
  var originalUriFormatter = tablePlugin.defaults.uriFormatter;

  tablePlugin.defaults.uriFormatter = function (uri, compact) {
    if (isSemPKMObjectIri(uri)) {
      var label = shortenIri(uri);
      var encodedIri = encodeURIComponent(uri);
      // Use onclick to call openTab — avoids full page navigation
      var safeIri = uri.replace(/'/g, "\\'");
      var safeLabel = label.replace(/'/g, "\\'");
      return '<a class="sparql-iri-link" href="/browser/object/' + encodedIri + '" ' +
        'onclick="event.preventDefault(); if(window.openTab){ window.openTab(\'' + safeIri + '\', \'' + safeLabel + '\'); } else { window.location.href = this.href; }" ' +
        'title="' + uri + '">' +
        label +
        '</a>';
    }
    // Fall through to default formatter
    if (typeof originalUriFormatter === 'function') {
      return originalUriFormatter(uri, compact);
    }
    return null; // Let YASR handle it with default rendering
  };

  return true;
}

// Register formatter before Yasgui instance is created (modifies global defaults)
// Call once the Yasgui script is loaded
function registerFormatterWhenReady() {
  if (!registerYasrUriFormatter()) {
    setTimeout(registerFormatterWhenReady, 100);
  }
}
registerFormatterWhenReady();
```

**Note on YASR plugin API:** The exact API for cell-level URI formatting varies between Yasgui versions. If `tablePlugin.defaults.uriFormatter` is not a recognized hook in v4.5.0, an alternative approach is to use a MutationObserver on `#yasgui-container` to post-process rendered table cells and wrap matching `<a>` tags. Include this fallback approach if the direct plugin hook does not work:

```js
// Fallback: MutationObserver post-processing
function postProcessYasrLinks(yasr) {
  var observer = new MutationObserver(function (mutations) {
    mutations.forEach(function (m) {
      m.addedNodes.forEach(function (node) {
        if (node.nodeType !== 1) return;
        var links = node.querySelectorAll ? node.querySelectorAll('td a[href]') : [];
        links.forEach(function (a) {
          var uri = a.getAttribute('href');
          if (isSemPKMObjectIri(uri) && !a.classList.contains('sparql-iri-link')) {
            a.className = 'sparql-iri-link';
            a.textContent = shortenIri(uri);
            a.title = uri;
            a.onclick = function (e) {
              e.preventDefault();
              if (window.openTab) window.openTab(uri, shortenIri(uri));
              else window.location.href = a.href;
            };
          }
        });
      });
    });
  });
  observer.observe(document.getElementById('yasgui-container'), {
    childList: true,
    subtree: true
  });
}
```
