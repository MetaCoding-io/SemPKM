**Script changes:**

Replace the entire IIFE script with this flow:

1. Define `containerId`, `formId`, `editorId`, `fallbackId`, `skeletonId` variables using the `{{ safe_id }}` template variable.

2. `initBodyEditor()` function — same logic as current: checks if `window.initEditor` exists, calls it with containerId, initialContent, objectIri. Returns boolean.

3. `enableFallback()` function — same as current but also:
   - Hide the skeleton element
   - Insert a `<div class="editor-fallback-message">Editor failed to load. Using plain text editor.</div>` before the fallback textarea
   - Show the fallback textarea (display: block)

4. `hideSkeleton()` function — sets `display: none` on the skeleton element.

5. `initVerticalSplit()` function — same as current: calls `Split()` with the two section IDs. Keep it as-is for now (cleanup registration comes in plan 10-03).

6. **Main initialization flow** — replace `setInterval` with:

```javascript
// Try immediate init (editor.js may already be loaded)
if (initBodyEditor()) {
  hideSkeleton();
} else {
  // Load editor module via dynamic import with 3-second timeout
  var timeout = new Promise(function(_, reject) {
    setTimeout(function() { reject(new Error('Editor load timeout')); }, 3000);
  });
  var load = import('/js/editor.js').then(function(mod) {
    if (mod && typeof mod.initEditor === 'function') {
      window.initEditor = mod.initEditor;
      window.editorAction = mod.editorAction;
      window.getEditor = mod.getEditor;
      window.destroyEditor = mod.destroyEditor;
    }
    return initBodyEditor();
  });
  Promise.race([load, timeout])
    .then(function(result) {
      hideSkeleton();
      if (!result) enableFallback();
    })
    .catch(function() {
      enableFallback();
    });
}
```

7. Call `initVerticalSplit()` at the end (outside the Promise chain — Split.js init is independent).

Also ensure the `<style>` block still hides `.body-fallback` by default (`display: none`). Keep all existing inline styles.

Important: Initialize CodeMirror BEFORE Split.js or in parallel — do NOT make CodeMirror init depend on Split.js. The `initVerticalSplit()` call should be after the editor init attempt but NOT inside the Promise chain, so Split.js doesn't block the editor.
