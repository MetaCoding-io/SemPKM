   ```python
   if "object_type" in filters and filters["object_type"]:
       # object_type is an IRI — match triples where subject OR object has this rdf:type
       # Since we're querying SQLite (not SPARQL), use a LIKE on the subject_iri
       # to match triples involving this type's instances.
       # Simpler approach: filter by subject_iri prefix matching the type IRI prefix,
       # OR better: the object_type filter matches the TYPE portion of the subject/object IRI.
       # Since subject_iri contains the full IRI like "urn:sempkm:model:basic-pkm:person-1",
       # we can filter by checking if the subject or object IRI contains the type slug.
       # Most robust: accept a type IRI and filter where subject_iri LIKE '%{type_slug}%'
       type_slug = filters["object_type"]
       stmt = stmt.where(
           (InferenceTripleState.subject_iri.contains(type_slug)) |
           (InferenceTripleState.object_iri.contains(type_slug))
       )
   if "date_from" in filters and filters["date_from"]:
       stmt = stmt.where(InferenceTripleState.inferred_at >= filters["date_from"])
   if "date_to" in filters and filters["date_to"]:
       # Add "T23:59:59" to include the full end day
       stmt = stmt.where(InferenceTripleState.inferred_at <= filters["date_to"] + "T23:59:59")
   ```

   ```python
   def _render_grouped_triples_html(triples, group_by: str):
       """Render triples grouped by the specified dimension."""
       from fastapi.responses import HTMLResponse
       from collections import defaultdict

       if not triples:
           return HTMLResponse(content='<div class="inference-empty">No inferred triples.</div>')

       groups = defaultdict(list)
       for t in triples:
           if group_by == "time":
               # Group by date portion of inferred_at
               key = t.get("inferred_at", "")[:10] or "Unknown"
           elif group_by == "object_type":
               # Extract type from subject IRI (last segment before the instance ID)
               subject = t.get("subject", "")
               key = _extract_type_from_iri(subject)
           elif group_by == "property_type":
               key = t.get("entailment_type", "Unknown")
           else:
               key = "All"
           groups[key].append(t)

       html_parts = []
       for group_key in sorted(groups.keys()):
           group_triples = groups[group_key]
           rows = _render_triple_rows(group_triples)
           html_parts.append(f'''<div class="inference-group">
     <h4 class="inference-group-header">{group_key} ({len(group_triples)})</h4>
     <table class="inference-triples-table">
       <thead><tr><th>Subject</th><th>Predicate</th><th>Object</th><th>Type</th><th>Status</th><th>Actions</th></tr></thead>
       <tbody>{rows}</tbody>
     </table>
   </div>''')

       return HTMLResponse(content="\n".join(html_parts))
   ```

   Add a small helper `_extract_type_from_iri()`:
   ```python
   def _extract_type_from_iri(iri: str) -> str:
       """Extract a type-like segment from an IRI for grouping."""
       # e.g., "urn:sempkm:model:basic-pkm:person-1" -> "person"
       parts = iri.rsplit(":", 1)
       if len(parts) == 2:
           # Remove trailing numeric ID: "person-1" -> "person"
           segment = parts[1]
           import re
           return re.sub(r'-\d+$', '', segment) or segment
       return _compact_iri(iri)
   ```
