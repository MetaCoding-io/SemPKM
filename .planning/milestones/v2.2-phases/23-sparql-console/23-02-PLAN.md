```js
var KNOWN_VOCAB_PREFIXES = [
  'http://www.w3.org/1999/02/22-rdf-syntax-ns#',
  'http://www.w3.org/2000/01/rdf-schema#',
  'http://www.w3.org/2002/07/owl#',
  'http://purl.org/dc/terms/',
  'http://www.w3.org/2004/02/skos/core#',
  'http://xmlns.com/foaf/0.1/',
  'https://schema.org/',
  'http://schema.org/',
  'http://www.w3.org/2001/XMLSchema#',
  'http://www.w3.org/ns/shacl#',
  'http://www.w3.org/ns/prov#',
  'urn:sempkm:',
];

function isSemPKMObjectIri(uri) {
  if (!uri) return false;
  // Use injected base namespace first (most accurate)
  var base = window.SEMPKM_BASE_NAMESPACE;
  if (base && uri.indexOf(base) === 0) return true;
  // Fallback: https:// IRI with path, not a known vocabulary
  if (uri.indexOf('https://') !== 0) return false;
  for (var i = 0; i < KNOWN_VOCAB_PREFIXES.length; i++) {
    if (uri.indexOf(KNOWN_VOCAB_PREFIXES[i]) === 0) return false;
  }
  try {
    var p = new URL(uri);
    return p.pathname.length > 1;
  } catch (e) { return false; }
}

function shortenSemPkmIri(uri) {
  var base = window.SEMPKM_BASE_NAMESPACE;
  if (base && uri.indexOf(base) === 0) {
    var local = uri.substring(base.length);
    return local.length > 50 ? local.substring(0, 47) + '...' : local;
  }
  try {
    var parts = new URL(uri).pathname.split('/').filter(Boolean);
    if (parts.length > 0) {
      var last = decodeURIComponent(parts[parts.length - 1]);
      return last.length > 50 ? last.substring(0, 47) + '...' : last;
    }
  } catch (e) {}
  return uri.length > 50 ? uri.substring(0, 47) + '...' : uri;
}

function registerYasrFormatter() {
  if (!window.Yasgui || !Yasgui.YASR || !Yasgui.YASR.plugins || !Yasgui.YASR.plugins.table) {
    return false;
  }
  var tablePlugin = Yasgui.YASR.plugins.table;
  if (!tablePlugin.defaults) tablePlugin.defaults = {};
  var origFormatter = tablePlugin.defaults.uriFormatter || null;
  tablePlugin.defaults.uriFormatter = function(uri, compact) {
    if (isSemPKMObjectIri(uri)) {
      var label = shortenSemPkmIri(uri);
      var safeUri = uri.replace(/\\/g, '\\\\').replace(/'/g, "\\'");
      var safeLabel = label.replace(/\\/g, '\\\\').replace(/'/g, "\\'");
      var href = '/browser/object/' + encodeURIComponent(uri);
      return '<a class="sparql-iri-link" href="' + href + '" ' +
        'onclick="event.preventDefault();if(window.openTab){window.openTab(\'' + safeUri + '\',\'' + safeLabel + '\');}else{window.location.href=this.href;}" ' +
        'title="' + uri.replace(/"/g, '&quot;') + '">' + label + '</a>';
    }
    return origFormatter ? origFormatter(uri, compact) : null;
  };
  return true;
}
```
