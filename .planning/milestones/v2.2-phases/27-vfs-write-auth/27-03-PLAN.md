Use a small inline script to handle the JSON POST (htmx encoding="application/json" sends form fields as JSON):
```html
<script>
function vfsSubmitTokenForm(e) {
  e.preventDefault();
  var name = document.getElementById('vfs-token-name').value.trim();
  if (!name) return false;
  fetch('/api/auth/tokens', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({name: name})
  }).then(function(r) { return r.json(); }).then(function(data) {
    document.getElementById('vfs-token-result').innerHTML =
      '<div class="vfs-token-reveal">' +
      '<p class="vfs-token-warning"><strong>Copy this token now.</strong> It will not be shown again.</p>' +
      '<div class="vfs-token-value-row">' +
      '<code class="vfs-token-value" id="vfs-new-token-value">' + data.token + '</code>' +
      '<button class="btn-icon" onclick="navigator.clipboard.writeText(\'' + data.token + '\'); this.textContent=\'Copied!\';">' +
      '<i data-lucide="copy" style="width:14px;height:14px;"></i></button>' +
      '</div></div>';
    // Append the new token row to the table without reloading
    var tbody = document.getElementById('vfs-tokens-tbody');
    if (tbody) {
      var tr = document.createElement('tr');
      tr.setAttribute('id', 'vfs-token-row-' + data.id);
      tr.innerHTML = '<td>' + data.name + '</td><td>' + new Date(data.created_at).toLocaleDateString() + '</td>' +
        '<td><button class="btn-danger-sm" onclick="vfsRevokeToken(\'' + data.id + '\', this)">Revoke</button></td>';
      tbody.appendChild(tr);
    }
    document.getElementById('vfs-token-name').value = '';
    if (typeof lucide !== 'undefined') lucide.createIcons();
  });
  return false;
}

function vfsRevokeToken(tokenId, btn) {
  if (!confirm('Revoke this token? Any WebDAV clients using it will be disconnected immediately.')) return;
  fetch('/api/auth/tokens/' + tokenId, {method: 'DELETE'}).then(function(r) {
    if (r.status === 204) {
      var row = document.getElementById('vfs-token-row-' + tokenId);
      if (row) row.remove();
    }
  });
}
</script>
```
