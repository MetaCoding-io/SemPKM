{# Jinja2 macro for rendering SHACL property groups as collapsible sections. Renders a group of properties wrapped in a
element (collapsible, expanded by default). Calls render_field for each property in the group. Usage: {% from "forms/_group.html" import render_group %} {{ render_group(group, properties, values) }} Args: group: PropertyGroup dataclass (iri, label, order) properties: list of PropertyShape items belonging to this group values: dict mapping property path -> current value(s) #} {% from "forms/_field.html" import render_field %} {% macro render_group(group, properties, values, ref_labels=None) %}
{{ group.label }}
{% for prop in properties %} {% set val = values.get(prop.path) if values else none %} {% if val is sequence and val is not string %} {{ render_field(prop, values=val, ref_labels=ref_labels) }} {% else %} {{ render_field(prop, value=val, ref_labels=ref_labels) }} {% endif %} {% endfor %}
{% endmacro %}
